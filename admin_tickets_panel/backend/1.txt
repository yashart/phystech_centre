    var annoucementSliderLoad = function() {
        jQuery('.generic_layout_container .js-main-slideshow li').removeAttr('style');

        var sliderClass = '.generic_layout_container .js-main-slideshow';

        var bxSlider = jQuery(sliderClass).bxSlider({ 
             controls: false,
             pager: true,
             adaptiveHeight: true,
             infiniteLoop: true    
        });
        bxSlider.startAuto();
        
        jQuery(sliderClass+' li.bx-clone > div').removeAttr('id'); /*bugfix - wuth last-slide slideToggle*/
        
        jQuery('.extended_view_more').click(function() {
            var height = jQuery(this).parents('li').find('.view_more_second').get( 0 ).scrollHeight 
                    + jQuery(this).parents('li').height() 
                    - jQuery(this).parents('li').find('.view_more_first')[0].scrollHeight;
            var originalHeight = jQuery(this).parents('.bx-viewport').height();
            jQuery(this).parents('.bx-viewport').attr('origHeight', originalHeight).animate({height: height}, 300);
        });
        jQuery('.extended_view_less').click(function() {
            var height = jQuery(this).parents('.bx-viewport').attr('origHeight');
            jQuery(this).parents('.bx-viewport').animate({height: height}, 300);
        });
        
        jQuery('.obj-main-slideshow').hover(function() {
            bxSlider.stopAuto();
        }, function() {
            bxSlider.startAuto();
        });
        jQuery(window).scroll(function() {
            var sliderTopPosition = bxSlider.offset().top;
            if( jQuery(this).scrollTop() > sliderTopPosition + 200 ) {
                bxSlider.stopAuto();
            } else {
                bxSlider.startAuto();
            }
        });
    };
